cmake_minimum_required(VERSION 3.16)
project(VisionFlow)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

if(NOT DEFINED OpenCV_DIR)
    set(OpenCV_DIR "C:/opencv/build/x64/vc16/lib")
endif()

find_package(OpenCV REQUIRED)

if(NOT OpenCV_FOUND)
    message(FATAL_ERROR "OpenCV not found! Make sure OpenCV_DIR is set correctly.")
endif()

# Include directories
include_directories(${OpenCV_INCLUDE_DIRS} include)

# Source files
file(GLOB_RECURSE SOURCES src/*.cpp src/stages/*.cpp src/core/*.cpp)

# Create executable
add_executable(VisionFlow ${SOURCES})

# Link libraries
target_link_libraries(VisionFlow ${OpenCV_LIBS})
